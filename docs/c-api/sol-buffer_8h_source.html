751 static inline int
752 sol_buffer_trim(struct sol_buffer *buf)
753 {
754  size_t new_size;
755 
756  if (!buf)
757  return -EINVAL;
758 
759  if (buf->flags & SOL_BUFFER_FLAGS_NO_NUL_BYTE)
760  new_size = buf->used;
761  else if (buf->used == SIZE_MAX)
762  return -EOVERFLOW;
763  else
764  new_size = buf->used + 1;
765 
766  if (new_size == buf->capacity)
767  return 0;
768 
769  return sol_buffer_resize(buf, new_size);
770 }
