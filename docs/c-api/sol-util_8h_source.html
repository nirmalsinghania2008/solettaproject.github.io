105 static inline void
106 sol_util_timespec_sum(const struct timespec *t1, const struct timespec *t2, struct timespec *result)
107 {
108  result->tv_nsec = t1->tv_nsec + t2->tv_nsec;
109  result->tv_sec = t1->tv_sec + t2->tv_sec;
110  if ((unsigned long long)result->tv_nsec >= SOL_NSEC_PER_SEC) {
111  result->tv_nsec -= SOL_NSEC_PER_SEC;
112  result->tv_sec++;
113  }

146 sol_util_timespec_compare(const struct timespec *t1, const struct timespec *t2)
147 {
148  int retval = (t1->tv_sec > t2->tv_sec) - (t1->tv_sec < t2->tv_sec);
