193 /* Power of 2 alignment */
194 #define DEFINE_ALIGN_POWER2(name_, type_, max_, clz_fn_) \
195  static inline type_ \
196  name_(type_ u) \
197  { \
198  unsigned int left_zeros; \
199  if (u <= 1) /* 0 is undefined for __builtin_clz() */ \
200  return u; \
201  left_zeros = clz_fn_(u - 1); \
202  if (unlikely(left_zeros <= 1)) \
203  return max_; \
204  return (type_)1 << ((sizeof(u) * 8) - left_zeros); \
205  }

282 static inline int
283 sol_util_ssize_mul(ssize_t op1, ssize_t op2, ssize_t *out)
284 {
285 #ifdef HAVE_BUILTIN_MUL_OVERFLOW
286  if (__builtin_mul_overflow(op1, op2, out))
287  return -EOVERFLOW;
288 #else
289  bool overflow = false;
290 
291  if (op1 > 0 && op2 > 0) {
292  overflow = op1 > OVERFLOW_SSIZE_T_POS / op2;
293  } else if (op1 > 0 && op2 <= 0) {
294  overflow = op2 < OVERFLOW_SSIZE_T_NEG / op1;
295  } else if (op1 <= 0 && op2 > 0) {
296  overflow = op1 < OVERFLOW_SSIZE_T_NEG / op2;
297  } else { // op1 <= 0 && op2 <= 0
298  overflow = op1 != 0 && op2 < OVERFLOW_SSIZE_T_POS / op1;
299  }
300 
301  if (overflow)
302  return -EOVERFLOW;
303 
304  *out = op1 * op2;
305 #endif
306  return 0;
307 }

408 sol_util_uuid_str_valid(const char *str)
409 {
410  size_t i, len;
411 
412  len = strlen(str);
413  if (len == 32) {
414  for (i = 0; i < len; i++) {
415  if (!isxdigit(str[i]))
416  return false;
417  }
418  } else if (len == 36) {
419  char c;
420  for (i = 0; i < len; i++) {
421  c = str[i];
422 
423  if (i == 8 || i == 13 || i == 18 || i == 23) {
424  if (c != '-')

426  } else if (!isxdigit(c))
427  return false;
428  }
429  } else
430  return false;
431 
432  return true;
