317 static inline struct sol_str_slice
318 sol_str_slice_remove_trailing_whitespace(struct sol_str_slice slice)
319 {
320  struct sol_str_slice copy = slice;
321 
322  while (copy.len != 0) {
323  char c = copy.data[copy.len - 1];
324  if (isspace(c))
325  copy.len--;
326  else
327  break;
328  }
329 
330  if (slice.len - copy.len)
331  return copy;
332 
333  return slice;
334 }
