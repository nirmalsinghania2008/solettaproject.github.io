59 accept4(int sockfd, struct sockaddr *addr, socklen_t *len, int flags)
60 {
61  int fl, fd = accept(sockfd, addr, len);
62 
63  if (fd < 0)
64  return fd;
65 
66  fl = fcntl(fd, F_GETFD);
67  if (fl == -1)
68  goto err;
69 
70  fl |= flags;
71  if (fcntl(fd, F_SETFD, fl) == -1)
72  goto err;
73 
74  return fd;
75 
76 err:
77  close(fd);
78  return -1;
79 }
