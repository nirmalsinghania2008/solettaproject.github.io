573 static inline bool
574 sol_json_loop_helper_generic(struct sol_json_scanner *scanner, struct sol_json_token *token,
575  enum sol_json_type end_type, enum sol_json_loop_reason *reason)
576 {
577  if (*reason != SOL_JSON_LOOP_REASON_OK)
578  return false;
579 
580  if (!sol_json_scanner_next(scanner, token)) {
581  *reason = SOL_JSON_LOOP_REASON_INVALID;
582  return false;
583  }
584 
585  if (sol_json_token_get_type(token) == end_type) {
586  *reason = SOL_JSON_LOOP_REASON_OK;
587  return false;
588  }
589 
590  if (sol_json_token_get_type(token) == SOL_JSON_TYPE_ELEMENT_SEP) {
591  if (!sol_json_scanner_next(scanner, token)) {
592  *reason = SOL_JSON_LOOP_REASON_INVALID;
593  return false;
594  }
595  }
596 
597  return true;
598 }
