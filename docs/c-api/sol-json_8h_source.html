169 static inline enum sol_json_type
170 sol_json_mem_get_type(const void *mem)
171 {
172  const char *p = (const char *)mem;
173 
174  if (strchr("{}[],:tfn\"", *p))
175  return (enum sol_json_type)*p;
176  if (isdigit(*p) || *p == '-' || *p == '+')
177  return SOL_JSON_TYPE_NUMBER;
178  return SOL_JSON_TYPE_UNKNOWN;
179 }

353 static inline bool
354 _sol_json_loop_helper_generic(struct sol_json_scanner *scanner, struct sol_json_token *token,
355  enum sol_json_type end_type, enum sol_json_loop_reason *reason)
356 {
357  if (*reason != SOL_JSON_LOOP_REASON_OK)
358  return false;
359 
360  if (!sol_json_scanner_next(scanner, token)) {
361  *reason = SOL_JSON_LOOP_REASON_INVALID;
362  return false;
363  }
364 
365  if (sol_json_token_get_type(token) == end_type) {
366  *reason = SOL_JSON_LOOP_REASON_OK;
367  return false;
368  }
369 
370  if (sol_json_token_get_type(token) == SOL_JSON_TYPE_ELEMENT_SEP) {
371  if (!sol_json_scanner_next(scanner, token)) {
372  *reason = SOL_JSON_LOOP_REASON_INVALID;
373  return false;
374  }
375  }
376 
377  return true;
378 }
