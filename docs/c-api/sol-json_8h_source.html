559 static inline bool
560 sol_json_loop_helper_generic(struct sol_json_scanner *scanner, struct sol_json_token *token,
561  enum sol_json_type end_type, enum sol_json_loop_reason *reason)
562 {
563  if (*reason != SOL_JSON_LOOP_REASON_OK)
564  return false;
565 
566  if (!sol_json_scanner_next(scanner, token)) {
567  *reason = SOL_JSON_LOOP_REASON_INVALID;
568  return false;
569  }
570 
571  if (sol_json_token_get_type(token) == end_type) {
572  *reason = SOL_JSON_LOOP_REASON_OK;
573  return false;
574  }
575 
576  if (sol_json_token_get_type(token) == SOL_JSON_TYPE_ELEMENT_SEP) {
577  if (!sol_json_scanner_next(scanner, token)) {
578  *reason = SOL_JSON_LOOP_REASON_INVALID;
579  return false;
580  }
581  }
582 
583  return true;
584 }
