333 sol_memmap_read_string(const char *name, char **value)
334 {
335  struct sol_buffer buf = SOL_BUFFER_INIT_EMPTY;
336  int r;
337 
338  r = sol_memmap_read_raw(name, &buf);
339  if (r < 0) {

342  }
343 
344  *value = sol_buffer_steal(&buf, NULL);
345 
346  return 0;
347 }

350 sol_memmap_write_string(const char *name, const char *value,
351  void (*cb)(void *data, const char *name, struct sol_blob *blob, int status),
352  const void *data)
353 {
354  int r;
355  struct sol_blob *blob;
356  char *string;
357 
358  string = strdup(value);
359  SOL_NULL_CHECK(string, -ENOMEM);
360 
361  blob = sol_blob_new(SOL_BLOB_TYPE_DEFAULT, NULL, string, strlen(value) + 1);
362  SOL_NULL_CHECK_GOTO(blob, error);
363 
364  r = sol_memmap_write_raw(name, blob, cb, data);
365  sol_blob_unref(blob);
366  return r;
367 
368 error:
369  free(string);
370 
371  return -ENOMEM;
372 }
