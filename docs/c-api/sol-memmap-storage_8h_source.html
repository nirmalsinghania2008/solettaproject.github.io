349 {

352

365 sol_memmap_write_string(const char *name, const char *value,
366  void (*cb)(void *data, const char *name, struct sol_blob *blob, int status),
367  const void *data)
368 {
369  int r;
370  struct sol_blob *blob;
371  char *string;
372 
373  string = strdup(value);
374  SOL_NULL_CHECK(string, -ENOMEM);
375 
376  blob = sol_blob_new(SOL_BLOB_TYPE_DEFAULT, NULL, string, strlen(value) + 1);
377  SOL_NULL_CHECK_GOTO(blob, error);
378 
379  r = sol_memmap_write_raw(name, blob, cb, data);
380  sol_blob_unref(blob);
381  return r;
382 
383 error:
384  free(string);
385 
386  return -ENOMEM;
