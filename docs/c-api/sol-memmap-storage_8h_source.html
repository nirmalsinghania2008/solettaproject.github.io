210 #define CREATE_BUFFER(_val) \
211  struct sol_buffer buf = SOL_BUFFER_INIT_FLAGS(_val, \
212  sizeof(*(_val)), SOL_BUFFER_FLAGS_MEMORY_NOT_OWNED | SOL_BUFFER_FLAGS_NO_NUL_BYTE);

217 #define CREATE_BLOB(_val) \
218  struct sol_blob *blob; \
219  size_t _s = sizeof(*_val); \
220  void *v = malloc(_s); \
221  SOL_NULL_CHECK(v, -ENOMEM); \
222  memcpy(v, _val, _s); \
223  blob = sol_blob_new(SOL_BLOB_TYPE_DEFAULT, NULL, v, _s); \
224  if (!blob) { \
225  free(v); \
226  return -EINVAL; \

361 static inline int
362 sol_memmap_read_string(const char *name, char **value)
363 {
364  struct sol_buffer buf = SOL_BUFFER_INIT_EMPTY;
365  int r;
366 
367  r = sol_memmap_read_raw(name, &buf);
368  if (r < 0) {
369  sol_buffer_fini(&buf);
370  return r;
371  }
372 
373  *value = sol_buffer_steal(&buf, NULL);
374 
375  return 0;
376 }

378 static inline int
379 sol_memmap_write_string(const char *name, const char *value,
380  void (*cb)(void *data, const char *name, struct sol_blob *blob, int status),
381  const void *data)
382 {
383  int r;
384  struct sol_blob *blob;
385  char *string;
386 
387  string = strdup(value);
388  SOL_NULL_CHECK(string, -ENOMEM);
389 
390  blob = sol_blob_new(SOL_BLOB_TYPE_DEFAULT, NULL, string, strlen(value) + 1);
391  SOL_NULL_CHECK_GOTO(blob, error);
392 
393  r = sol_memmap_write_raw(name, blob, cb, data);
394  sol_blob_unref(blob);
395  return r;
396 
397 error:
398  free(string);
399 
400  return -ENOMEM;
401 }
